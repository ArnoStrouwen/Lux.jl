334 combinedef(dict::Dict)
335
336 `combinedef` is the inverse of [`splitdef`](@ref). It takes a `splitdef`-like Dict
337 and returns a function definition.
338
339 This function approximately does the following (but more sophisticated to avoid
340 emitting parts that did not actually appear in the original function definition.)
341
342 ```julia
343 rtype = get(dict, :rtype, :Any)
344 all_params = [get(dict, :params, [])..., get(dict, :whereparams, [])...]
345 :(function \$(dict[:name]){\$(all_params...)}(\$(dict[:args]...);
346 \$(dict[:kwargs]...))::\$rtype
347 \$(dict[:body])
348 end)
349 ```
350
351 """
352 function combinedef(dict::Dict)
353 rtype = get(dict, :rtype, nothing)
354 params = get(dict, :params, [])
355 wparams = get(dict, :whereparams, [])
356 body = block(dict[:body])
357 if haskey(dict, :name)
358 name = dict[:name]
359 name_param = isempty(params) ? name : :($name{$(params...)})
360 # We need the `if` to handle parametric inner/outer constructors like
361 # SomeType{X}(x::X) where X = SomeType{X}(x, x+2)
362 if isempty(wparams)
363 if rtype==nothing
364 @q(function $name_param($(dict[:args]...);
365 $(dict[:kwargs]...))
366 $(body.args...)
367 end)
368 else
369 @q(function $name_param($(dict[:args]...);
370 $(dict[:kwargs]...))::$rtype
371 $(body.args...)
372 end)
373 end
374 else
375 if rtype==nothing
376 @q(function $name_param($(dict[:args]...);
377 $(dict[:kwargs]...)) where {$(wparams...)}
378 $(body.args...)
379 end)
380 else
381 @q(function $name_param($(dict[:args]...);
382 $(dict[:kwargs]...))::$rtype where {$(wparams...)}
383 $(body.args...)
384 end)
385 end
386 end
387 else
388 if isempty(dict[:kwargs])
389 arg = :($(dict[:args]...),)
390 else
391 arg = Expr(:tuple, Expr(:parameters, dict[:kwargs]...), dict[:args]...)
392 end
393 if isempty(wparams)
394 if rtype==nothing
395 @q($arg -> $body)
396 else
397 @q(($arg::$rtype) -> $body)
398 end
399 else
400 if rtype==nothing
401 @q(($arg where {$(wparams...)}) -> $body)
402 else
403 @q(($arg::$rtype where {$(wparams...)}) -> $body)
404 end
405 end
406 end
