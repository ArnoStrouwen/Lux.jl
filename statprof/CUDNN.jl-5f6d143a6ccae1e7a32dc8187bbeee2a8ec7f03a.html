132 function _log_message(sev, dbg, str)
133 lines = split(str, '\0')
134 msg = join(lines, '\n')
135 if sev == CUDNN_SEV_INFO
136 @debug msg
137 elseif sev == CUDNN_SEV_WARNING
138 @warn msg
139 elseif sev == CUDNN_SEV_ERROR
140 @error msg
141 elseif sev == CUDNN_SEV_FATAL
142 error(msg)
143 end
144 return
145 end
