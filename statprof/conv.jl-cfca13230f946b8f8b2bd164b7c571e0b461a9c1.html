1 ## Convolution API
2 #
3 # We provide the following generic methods, for 3d, 4d, and 5d tensors, calculating 1d,
4 # 2d and 3d convolutions, based on the rank of the input tensors, in both mutating and
5 # non-mutating auto-allocating variants:
6 # - Convolution:
7 # - conv(x, w, cdims)
8 # - conv!(y, x, w, cdims)
9 # - Convolution data backpropagation
10 # - ∇conv_data(dy, w, cdims)
11 # - ∇conv_data!(dx, dy, w, cdims)
12 # - Convolution filter backpropagation
13 # - ∇conv_filter(x, dy, cdims)
14 # - ∇conv_filter!(dw, x, dy, cdims)
15 #
16 # All methods require a `ConvDims` object to define the dimensions and optional
17 # elements of the convolution (padding, stride, dilation, kernel-flipping, etc...),
18 # which is easily constructable through something like `DenseConvDims(x, w)`. All
19 # methods take in the `ConvDims` of the associated normal, forward-pass convolution,
20 # that is, the following is legal:
21 #
22 # cdims = ConvDims(x, w; stride=2, dilation=(3,2))
23 # dx = ∇conv_data(conv(x, w, cdims), w, cdims)
24
25 # The computational flow, starting from the user facing functions,
26 # goes through the following steps:
27 #
28 # STEP 1:
29 # use ConvDims objects (only for `conv` and `depthwiseconv`)
30 # STEP 2:
31 # define autoallocating version (frontend and implementations)
32 # STEP 3:
33 # reshape to 3d convolutions (frontend and implementions)
34 # STEP 4:
35 # choose implementation
36
37 # TODO: should we also add
38 # STEP X:
39 # use homogeneus datatypes
40 # to handle etherogeneus inputs now handled by conv_direct?

44 """
45 conv(x, w; stride = 1, pad = 0, dilation = 1, flipped = false, groups = 1)
46
47 Apply convolution filter `w` to input `x`. `x` and `w` are 3d/4d/5d tensors
48 in 1d/2d/3d convolutions respectively.
49 """
50 function conv(x, w::AbstractArray{T, N}; stride = 1, pad = 0, dilation = 1, flipped = false, groups = 1) where {T, N}
51 stride = expand(Val(N - 2), stride)
52 padding = expand(Val(N - 2), pad)
53 dilation = expand(Val(N - 2), dilation)
54 cdims = DenseConvDims(
55 size(x), size(w); stride, padding, dilation, flipkernel=flipped, groups)
56 return conv(x, w, cdims)
57 end
58
59 """
60 depthwiseconv(x, w; stride=1, pad=0, dilation=1, flipped=false)
61
62 Depthwise convolution operation with filter `w` on input `x`. `x` and `w`
63 are 3d/4d/5d tensors in 1d/2d/3d convolutions respectively.
64 """
65 function depthwiseconv(x, w::AbstractArray{T, N}; stride=1, pad=0, dilation=1, flipped=false) where {T, N}
66 stride = expand(Val(N-2), stride)
67 pad = expand(Val(N-2), pad)
68 dilation = expand(Val(N-2), dilation)
69 cdims = DepthwiseConvDims(x, w; stride=stride, padding=pad, dilation=dilation, flipkernel=flipped)
70 return depthwiseconv(x, w, cdims)
71 end
