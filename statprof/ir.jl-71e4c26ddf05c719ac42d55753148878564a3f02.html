198 """
199 block!(ir::IR)
200 block!(ir::IR, i)
201
202 Insert a new block in `ir`. If `i` is given, the block is inserted at this position, otherwise it
203 is appended at the end. Branches in all other blocks are updated to preserve the original
204 behaviour.
205 """
206 function block!(ir::IR, i = length(blocks(ir))+1)
207 insert!(ir.blocks, i, BasicBlock())
208 if i != length(blocks(ir))
209 for b in blocks(ir), bi = 1:length(branches(b))
210 br = branches(b)[bi]
211 br.block >= i && (branches(b)[bi] = Branch(br, block = br.block+1))
212 end
213 for (ii, (b, j)) = enumerate(ir.defs)
214 b >= i && (ir.defs[ii] = (b+1, j))
215 end
216 end
217 return block(ir, i)
218 end

279 Convert implicit fallthroughs to explicit branches to the next block (these occur when the last
280 branch in a block is conditional):
