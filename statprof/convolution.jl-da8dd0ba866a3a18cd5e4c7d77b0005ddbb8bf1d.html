9
10 Return the convolution of filter `w` with tensor `x`, overwriting `y` if provided, according
11 to keyword arguments or the convolution descriptor `d`. Optionally perform bias addition,
12 activation and/or scaling:
13
14 y .= activation.(alpha * conv(w,x) + beta * z .+ bias)
15
16 All tensors should have the same number of dimensions. If they are less than 4-D their
17 dimensions are assumed to be padded on the left with ones. `x` has size `(X...,Cx,N)` where
18 `(X...)` are the spatial dimensions, `Cx` is the number of input channels, and `N` is the
19 number of instances. `y,z` have size `(Y...,Cy,N)` where `(Y...)` are the spatial dimensions
20 and `Cy` is the number of output channels (`y` and `z` can be the same array). Both `Cx` and
21 `Cy` have to be an exact multiple of `group`. `w` has size `(W...,Cx√∑group,Cy)` where
22 `(W...)` are the filter dimensions. `bias` has size `(1...,Cy,1)`.
23
24 The arguments `padding`, `stride` and `dilation` can be specified as `n-2` dimensional
25 vectors, tuples or a single integer which is assumed to be repeated `n-2` times. If any of
26 the entries is larger than the corresponding `x` dimension, the `x` dimension is used
27 instead. For a description of different types of convolution see:
28 https://towardsdatascience.com/a-comprehensive-introduction-to-different-types-of-convolutions-in-deep-learning-669281e58215
29
30 Keyword arguments:
31 * `activation = CUDNN_ACTIVATION_IDENTITY`: the only other supported option is `CUDNN_ACTIVATION_RELU`
32 * `bias = nothing`: add bias if provided
33 * `z = nothing`: add `beta*z`, `z` can be `nothing`, `y` or another array similar to `y`
34 * `alpha = 1, beta = 0`: scaling parameters
35 * `format = CUDNN_TENSOR_NCHW`: order of tensor dimensions, the other alternative is `CUDNN_TENSOR_NHWC`. Note that Julia dimensions will have the opposite order, i.e. WHCN or CWHN.
36
37 Keyword arguments describing the convolution when `d` is not given:
38 * `mode = CUDNN_CONVOLUTION`: alternatively `CUDNN_CROSS_CORRELATION`
39 * `padding = 0`: padding assumed around `x`
40 * `stride = 1`: how far to shift the convolution window at each step
41 * `dilation = 1`: dilation factor
42 * `group = 1`: number of groups to be used
43 * `mathType = CUDNN.math_mode()`: whether or not the use of tensor op is permitted
44 * `reorderType = CUDNN_DEFAULT_REORDER`: convolution reorder type

157 # Convert the integer, tuple or array to padding/stride/dilation dims compatible with array size
158 function convdims(d, s::Dims{N}, default) where N
159 @assert d isa Integer || length(d) == N-2 "Cannot use $d padding/stride/dilation with $(Base.dims2string(s)) array."
160 if N >= 4
161 (d isa Integer ? fill(Cint(d), N-2) : Cint[reverse(d)...])
162 elseif N == 3
163 Cint[d[1], default] # 3D tensors are padded to 4D
164 else
165 Cint[default, default] # lower dim tensors have no spatial dims

224 # Return algorithm with best memory that is within 10% of best time
225 function cudnnConvolutionAlgoPerfChoose(ps, n)
226 (ibest,mbest,tbest) = (0,Inf,Inf)
227 for i in 1:n
228 # These metrics are written in a sorted fashion where the first element has the lowest compute time.
229 if ps[i].status == CUDNN_STATUS_SUCCESS && ps[i].memory < mbest && ps[i].time < tbest * 1.1
230 (ibest,mbest,tbest) = (i,ps[i].memory,ps[i].time)
231 end
232 end
233 if ibest == 0
234 @warn "No valid algorithm found, probably bad params for convolution." maxlog=1
235 ibest = findfirst(p->p.algo==0, ps)
236 ibest === nothing && error("Cannot find backup algorithm for convolution, giving up.")
237 end
238 return ps[ibest]
239 end
