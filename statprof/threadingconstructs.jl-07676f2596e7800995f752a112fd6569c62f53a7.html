25 function threading_run(func)
26 ccall(:jl_enter_threaded_region, Cvoid, ())
27 n = nthreads()
28 tasks = Vector{Task}(undef, n)
29 for i = 1:n
30 t = Task(func)
31 t.sticky = true
32 ccall(:jl_set_task_tid, Cvoid, (Any, Cint), t, i-1)
33 tasks[i] = t
34 schedule(t)
35 end
36 try
37 for i = 1:n
38 wait(tasks[i])
39 end
40 finally
41 ccall(:jl_exit_threaded_region, Cvoid, ())
42 end
43 end

45 function _threadsfor(iter, lbody, schedule)
46 lidx = iter.args[1] # index
47 range = iter.args[2]
48 quote
49 local threadsfor_fun
50 let range = $(esc(range))
51 function threadsfor_fun(onethread=false)
52 r = range # Load into local variable
53 lenr = length(r)
54 # divide loop iterations among threads
55 if onethread
56 tid = 1
57 len, rem = lenr, 0
58 else
59 tid = threadid()
60 len, rem = divrem(lenr, nthreads())
61 end
62 # not enough iterations for all the threads?
63 if len == 0
64 if tid > rem
65 return
66 end
67 len, rem = 1, 0
68 end
69 # compute this thread's iterations
70 f = firstindex(r) + ((tid-1) * len)
71 l = f + len - 1
72 # distribute remaining iterations evenly
73 if rem > 0
74 if tid <= rem
75 f = f + (tid-1)
76 l = l + tid
77 else
78 f = f + rem
79 l = l + rem
80 end
81 end
82 # run this thread's iterations
83 for i = f:l
84 local $(esc(lidx)) = @inbounds r[i]
85 $(esc(lbody))
86 end
87 end
88 end
89 if threadid() != 1 || ccall(:jl_in_threaded_region, Cint, ()) != 0
90 $(if schedule === :static
91 :(error("`@threads :static` can only be used from thread 1 and not nested"))
92 else
93 # only use threads when called from thread 1, outside @threads
94 :(Base.invokelatest(threadsfor_fun, true))
95 end)
96 else
97 threading_run(threadsfor_fun)
98 end
99 nothing
100 end
101 end

103 """
104 Threads.@threads [schedule] for ... end
105
106 A macro to parallelize a `for` loop to run with multiple threads. Splits the iteration
107 space among multiple tasks and runs those tasks on threads according to a scheduling
108 policy.
109 A barrier is placed at the end of the loop which waits for all tasks to finish
110 execution.
111
112 The `schedule` argument can be used to request a particular scheduling policy.
113 The only currently supported value is `:static`, which creates one task per thread
114 and divides the iterations equally among them. Specifying `:static` is an error
115 if used from inside another `@threads` loop or from a thread other than 1.
116
117 The default schedule (used when no `schedule` argument is present) is subject to change.
118
119 !!! compat "Julia 1.5"
120 The `schedule` argument is available as of Julia 1.5.
121
122 See also: [`@spawn`](@ref Threads.@spawn), [`nthreads()`](@ref Threads.nthreads),
123 [`threadid()`](@ref Threads.threadid), `pmap` in [`Distributed`](@ref man-distributed),
124 `BLAS.set_num_threads` in [`LinearAlgebra`](@ref man-linalg).
125 """
126 macro threads(args...)
127 na = length(args)
128 if na == 2
129 sched, ex = args
130 if sched isa QuoteNode
131 sched = sched.value
132 elseif sched isa Symbol
133 # for now only allow quoted symbols
134 sched = nothing
135 end
136 if sched !== :static
137 throw(ArgumentError("unsupported schedule argument in @threads"))
138 end
139 elseif na == 1
140 sched = :default
141 ex = args[1]
142 else
143 throw(ArgumentError("wrong number of arguments in @threads"))
144 end
145 if !(isa(ex, Expr) && ex.head === :for)
146 throw(ArgumentError("@threads requires a `for` loop expression"))
147 end
148 if !(ex.args[1] isa Expr && ex.args[1].head === :(=))
149 throw(ArgumentError("nested outer loops are not currently supported by @threads"))
150 end
151 return _threadsfor(ex.args[1], ex.args[2], sched)
152 end

154 """
155 Threads.@spawn expr
156
157 Create a [`Task`](@ref) and [`schedule`](@ref) it to run on any available thread.
158 The task is allocated to a thread after it becomes available. To wait for the task
159 to finish, call [`wait`](@ref) on the result of this macro, or call [`fetch`](@ref) to
160 wait and then obtain its return value.
161
162 Values can be interpolated into `@spawn` via `\$`, which copies the value directly into the
163 constructed underlying closure. This allows you to insert the _value_ of a variable,
164 isolating the asynchronous code from changes to the variable's value in the current task.
165
166 !!! note
167 See the manual chapter on threading for important caveats.
168
169 !!! compat "Julia 1.3"
170 This macro is available as of Julia 1.3.
171
172 !!! compat "Julia 1.4"
173 Interpolating values via `\$` is available as of Julia 1.4.
174 """
